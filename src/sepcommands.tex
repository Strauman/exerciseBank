\def\exenv@endcurrent#1{
  \@ifundefined{exenv@current}{
    \edef\tmp{\csname exenv@end#1\endcsname}

  }{}
}
\def\exenv@begin#1{
  \gdef\exenv@current{#1}
}
\def\exenv@separator#1{
  % \exenv@endcurrent
  \edef\tmp{\csname exenv@begin#1\endcsname}
\tmp
}
